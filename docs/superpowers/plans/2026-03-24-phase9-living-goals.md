- [ ] **Step 7: Implement update_goal_progress in executeTool**

In `src/services/ai.ts`, add a new case in `executeTool` after `case "set_goal"`:

```typescript
    case "update_goal_progress": {
      const input = toolInput as {
        goalId: number;
        progress?: number;
        currentValue?: number;
        note?: string;
      };

      const goal = await prisma.goal.findFirst({
        where: { id: input.goalId, userId, status: "active" },
      });

      if (!goal) {
        return { text: `Цель с ID ${input.goalId} не найдена или неактивна.`, actions: [] };
      }

      const updateData: Record<string, unknown> = {};

      if (input.currentValue !== undefined) {
        updateData.currentValue = input.currentValue;
        // Auto-calculate progress if targetValue exists
        if (goal.targetValue && goal.targetValue > 0) {
          updateData.progress = Math.min(100, Math.round((input.currentValue / goal.targetValue) * 100));
        }
      }

      if (input.progress !== undefined) {
        updateData.progress = Math.min(100, Math.max(0, input.progress));
      }

      if (Object.keys(updateData).length === 0) {
        return { text: "Нет данных для обновления прогресса.", actions: [] };
      }

      // Auto-complete if progress reaches 100
      if ((updateData.progress as number) >= 100) {
        updateData.status = "completed";
      }

      await prisma.goal.update({ where: { id: goal.id }, data: updateData });

      const AREA_LABELS: Record<string, string> = {
        health: "Здоровье", career: "Карьера", relationships: "Отношения",
        finances: "Финансы", family: "Семья", growth: "Развитие",
        recreation: "Отдых", environment: "Среда",
      };

      const areaLabel = AREA_LABELS[goal.lifeArea] || goal.lifeArea;
      const progressStr = updateData.progress !== undefined ? ` — ${updateData.progress}%` : "";
      const valueStr = updateData.currentValue !== undefined && goal.metric
        ? ` (${updateData.currentValue}${goal.targetValue ? `/${goal.targetValue}` : ""} ${goal.metric})`
        : "";
      const completedStr = (updateData.progress as number) >= 100 ? " 🎉 Цель достигнута!" : "";

      return {
        text: `Прогресс "${goal.title}" (${areaLabel}) обновлён${progressStr}${valueStr}.${completedStr}${input.note ? ` Заметка: ${input.note}` : ""}`,
        actions: [],
      };
    }
```

function GoalLink({ goalId }: { goalId: number }) {

    api.goals("active").then(goals => {
      const found = goals.find(g => g.id === goalId);
      if (found) setGoal({ title: found.title, progress: found.progress });
    }).catch(() => {});

  if (!goal) return null;

  return (
    <div class="goal-link-info">
      <div class="goal-link-title">{goal.title}</div>
      <div class="goal-link-progress-bar">
        <div class="goal-link-progress-fill" style={{ width: `${goal.progress}%` }} />
      </div>

/**
 * Weekly target consistency: completions this week / targetPerWeek * 100.
 * Used when frequency is "weekly" with a targetPerWeek > 1.
 */
export function calculateWeeklyTargetConsistency(
  logs: { date: Date }[],
  targetPerWeek: number,
  today: Date = new Date(),
): number {
  if (targetPerWeek <= 0) return 0;

  const todayMidnight = new Date(today);
  todayMidnight.setHours(0, 0, 0, 0);

  // Find start of current week (Monday)
  const dow = todayMidnight.getDay();
  const mondayOffset = dow === 0 ? 6 : dow - 1;
  const weekStart = new Date(todayMidnight);
  weekStart.setDate(weekStart.getDate() - mondayOffset);

  const completedThisWeek = logs.filter(l => {
    const d = new Date(l.date);
    d.setHours(0, 0, 0, 0);
    return d >= weekStart && d <= todayMidnight;
  }).length;

  return Math.min(100, Math.round((completedThisWeek / targetPerWeek) * 100));
}

      const digest = await prisma.weeklyDigest.findUnique({
        where: {
          userId_weekStart: { userId, weekStart },
        },
      });

      if (!digest) {
        return res.status(404).json({ error: "Digest not found" });
      }

      res.json({
        id: digest.id,
        weekStart: digest.weekStart.toISOString(),
        content: digest.content,
        summary: digest.summary,
        createdAt: digest.createdAt.toISOString(),
      });
    } catch (err) {
      console.error("[digests] GET by week error:", err);
      res.status(500).json({ error: "Failed to get digest" });
    }
  });
}
